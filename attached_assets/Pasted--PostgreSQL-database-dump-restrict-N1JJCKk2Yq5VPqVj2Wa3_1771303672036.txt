--
-- Name: ensure_primary_domain_exists(); Type: FUNCTION; Schema: public; Owner: -
--

CREATE FUNCTION public.ensure_primary_domain_exists() RETURNS trigger
    LANGUAGE plpgsql
    AS $$
BEGIN
  IF OLD.is_primary = true THEN
    IF NOT EXISTS (
      SELECT 1 FROM venue_domains
      WHERE venue_id = OLD.venue_id
      AND id <> OLD.id
      AND is_primary = true
    ) THEN
      UPDATE venue_domains
      SET is_primary = true
      WHERE id = (
        SELECT id FROM venue_domains
        WHERE venue_id = OLD.venue_id
        AND id <> OLD.id
        LIMIT 1
      );
      RAISE WARNING '[venue_domains_ensure_primary] Trigger fired: auto-promoted replacement primary for venue_id=% after deletion of domain_id=%', OLD.venue_id, OLD.id;
    END IF;
  END IF;
  RETURN OLD;
END;
$$;
