CREATE server/routes/we-whitelabel.ts
Register: app.use('/api/we/whitelabel', weAuth, weVenue,
  requireWhiteLabel)

ROUTES:

GET /api/we/whitelabel
- Returns current white label config for venueId
- { agencyName, agencyLogo, brandColours,
  stagingDomain, smtpConfigured, fromName, fromEmail }
- Never returns SMTP password or credentials

POST /api/we/whitelabel/config
- Updates agency name, logo, brand colours
- Stores in workspace settings
- Logs to we_audit_log

POST /api/we/whitelabel/smtp
- Body: { host, port, username, password,
  fromName, fromEmail }
- Validates all fields present
- Stores credentials in HashiCorp Vault
  keyed by venueId
- Does NOT save until test passes
- Triggers test email automatically before saving
- Returns { testSent: true } — client must confirm
  test received before credentials saved

POST /api/we/whitelabel/smtp/confirm
- Called after agency confirms test email received
- Saves SMTP credentials to Vault
- Logs to we_audit_log: action 'smtp_configured'
  never logs credentials

POST /api/we/whitelabel/smtp/test
- Sends test email using current stored credentials
- To: authenticated user's email address
- Subject: 'Test email from [agency name]'
- Returns { sent: true } or { error: string }

DELETE /api/we/whitelabel/smtp
- Removes SMTP credentials from Vault
- Reverts to IndexFlow email domain
- Logs to we_audit_log

POST /api/we/whitelabel/staging-domain
- Body: { domain } e.g. staging.boltagency.com
- Validates domain format
- Returns DNS configuration instructions:
  { cnameRecord, cnameTarget, txtRecord, txtValue }
- Stores pending staging domain
- Does not activate until DNS verified

POST /api/we/whitelabel/staging-domain/verify
- Checks CNAME record points to IndexFlow
- On success: activates staging domain
- All existing test URLs migrate to new staging domain
- Logs to we_audit_log

DELETE /api/we/whitelabel/staging-domain
- Removes custom staging domain
- Reverts all URLs to indexflow.cloud
- Logs to we_audit_log

Register route in server/routes.ts
